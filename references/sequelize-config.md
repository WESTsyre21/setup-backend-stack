## `config/config.js`

Replace the default `config/config.json` generated by `sequelize-cli init` with a
JavaScript version so it can read from environment variables:

```js
require('dotenv').config();

module.exports = {
  development: {
    username: process.env.DB_USER,
    password: process.env.DB_PASSWORD,
    database: process.env.DB_NAME,
    host:     process.env.DB_HOST,
    port:     parseInt(process.env.DB_PORT, 10),
    dialect:  'postgres',
    logging:  false,
  },
  test: {
    username: process.env.DB_USER,
    password: process.env.DB_PASSWORD,
    database: `${process.env.DB_NAME}_test`,
    host:     process.env.DB_HOST,
    port:     parseInt(process.env.DB_PORT, 10),
    dialect:  'postgres',
    logging:  false,
  },
  production: {
    use_env_variable: 'DATABASE_URL',
    dialect:          'postgres',
    dialectOptions: {
      ssl: { require: true, rejectUnauthorized: false },
    },
  },
};
```

Update `.sequelizerc` to point at the JS config:

```js
const path = require('path');

module.exports = {
  config:           path.resolve('config', 'config.js'),
  'models-path':    path.resolve('models'),
  'seeders-path':   path.resolve('seeders'),
  'migrations-path': path.resolve('migrations'),
};
```

## `server.js` snippet

```js
const express    = require('express');
const { Sequelize } = require('sequelize');
require('dotenv').config();

const sequelize = new Sequelize(
  process.env.DB_NAME,
  process.env.DB_USER,
  process.env.DB_PASSWORD,
  {
    host:    process.env.DB_HOST,
    port:    parseInt(process.env.DB_PORT, 10),
    dialect: 'postgres',
    logging: false,
  }
);

const app  = express();
const PORT = process.env.PORT || 3000;

app.get('/health', async (req, res) => {
  try {
    await sequelize.authenticate();
    res.status(200).json({ status: 'ok' });
  } catch (err) {
    res.status(500).json({ status: 'error', message: err.message });
  }
});

app.get('/', (_req, res) => res.send('Backend is running.'));

app.listen(PORT, () => console.log(`Server on http://localhost:${PORT}`));
```

## Initial model

Generate a `User` model as the baseline:

```bash
npx sequelize-cli model:generate \
  --name User \
  --attributes email:string,name:string
```

This creates:
- `models/user.js`
- `migrations/<timestamp>-create-user.js`

Edit the migration to add `unique: true` on `email` and `allowNull: false`:

```js
// Inside the migration's `up` function, replace the email column:
email: {
  type: Sequelize.STRING,
  allowNull: false,
  unique: true,
},
```

Run the migration:

```bash
npx sequelize-cli db:migrate
```

## Baseline migration (existing DB)

If `SequelizeMeta` already exists, do **not** run `db:migrate` from scratch.

1. Create a new migration file that matches the current schema state:

```bash
npx sequelize-cli migration:generate --name baseline
```

2. Leave the `up` function empty (the schema already exists) and the `down`
   function empty too (you don't want to roll back existing data):

```js
'use strict';

module.exports = {
  async up(queryInterface, Sequelize) {
    // Schema already applied — this is a baseline marker only.
  },
  async down(queryInterface, Sequelize) {
    // No-op rollback.
  },
};
```

3. Insert the baseline record into `SequelizeMeta`:

```bash
npx sequelize-cli db:migrate
```

Future migrations will now stack on top of the baseline.
